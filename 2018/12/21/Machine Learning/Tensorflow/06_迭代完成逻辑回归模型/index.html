# 所有样本迭代次数
training_epochs = 50
# 每进行一次迭代需要多少样本
batch_size      = 100
display_step    = 5

# Session
# 上面有初始化 init = tf.global_variables_initializer()
sess = tf.Session()
sess.run(init)

# Mini-batch learning
for epoch in range(training_epochs):
    # 对于每一个 epoch 初始的损失值都为 0
    avg_cost = 0.
    num_batch = int(mnist.train.num_examples/batch_size)
    for i in range(num_batch):
        # 通过 next_batch 返回相应的 batch_xs，batch_ys
        batch_xs, batch_ys       = mnist.train.next_batch(batch_size)
        # run 梯度下降求解
        # 初始化 x y 的时候用的 placeholder，开始时还没赋值
        #　由于是一个 batch 一个 batch 取的
        # tf 规定用字典的形式将 values 赋值给 keys
        sess.run(optm, feed_dict = {x: batch_xs, y: batch_ys})
        feeds                    = {x: batch_xs, y: batch_ys}
        # 计算损失值，目的是打印出来看一下结果
        avg_cost                += sess.run(cost, feed_dict=feeds)/num_batch
    # Display
    # 每 5 个 epoch print 一次
    if epoch % display_step == 0:
        # feeds_train 和 feeds_test 是字典结构
        feeds_train  = {x: batch_xs, y: batch_ys}
        feeds_test   = {x: mnist.test.images, y: mnist.test.labels}
        train_acc    = sess.run(accr, feed_dict=feeds_train)
        test_acc     = sess.run(accr, feed_dict=feeds_test)
        print("Epoch: %03d/%03d cost: %.9f train_acc: %.3f test_acc: %.3f"
             % (epoch, training_epochs, avg_cost, train_acc, test_acc))
print("DONE")
